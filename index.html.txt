<script>
const products=[
  {id:1,title:"Sneakers",price:1599,img:"https://images.unsplash.com/photo-1600180758898-11b6b05a6f1b?w=600"},
  {id:2,title:"Backpack",price:899,img:"https://images.unsplash.com/photo-1542291026-7eec264c27ff?w=600"},
  {id:3,title:"Earbuds",price:1299,img:"https://images.unsplash.com/photo-1542751371-adc38448a05e?w=600"},
  {id:4,title:"Ceramic Mug",price:249,img:"https://images.unsplash.com/photo-1509042239860-f550ce710b93?w=600"},
];
let cart={};
function renderProducts(){
  const wrap=document.getElementById("products");
  wrap.innerHTML="";
  products.forEach(p=>{
    wrap.innerHTML+=`
    <div class="card">
      <img src="${p.img}" alt="${p.title}">
      <h3>${p.title}</h3>
      <p>High quality ${p.title.toLowerCase()}.</p>
      <div class="price-row">
        <span class="price">₱${p.price}</span>
        <button class="btn btn-primary" onclick="addToCart(${p.id})">Add</button>
      </div>
    </div>`;
  });
}
function addToCart(id){
  cart[id]=(cart[id]||0)+1;
  renderCart();
}
function renderCart(){
  const items=document.getElementById("cart-items");
  const empty=document.getElementById("cart-empty");
  items.innerHTML="";
  let total=0,count=0;
  for(let id in cart){
    const p=products.find(x=>x.id==id);
    const qty=cart[id];
    total+=p.price*qty; count+=qty;
    items.innerHTML+=`
      <div class="item">
        <img class="thumb" src="${p.img}">
        <div class="meta"><b>${p.title}</b><span class="qty">₱${p.price} × ${qty}</span></div>
        <div>
          <button class="btn-ghost btn" onclick="changeQty(${id},-1)">-</button>
          <button class="btn-ghost btn" onclick="changeQty(${id},1)">+</button>
        </div>
      </div>`;
  }
  document.getElementById("cart-count").textContent=count;
  document.getElementById("cart-total").textContent="₱"+total.toFixed(2);
  empty.style.display=count? "none":"block";
}
function changeQty(id,delta){
  cart[id]+=delta;
  if(cart[id]<=0) delete cart[id];
  renderCart();
}


// ===== Modal + Verification =====
function showModal(content){
  document.getElementById("modal-content").innerHTML=content;
  document.getElementById("modal").style.display="flex";
}
function closeModal(){document.getElementById("modal").style.display="none";}


function verifyAndCheckout(mode){
  // Step 1: show loading spinner
  showModal(`
    <h3>Verifying your order...</h3>
    <div class="spinner"></div>
    <p>Please wait a moment</p>
  `);


  // Step 2: after 3 seconds, show checkout form
  setTimeout(()=>{
    showModal(`
      <h3>${mode==="now"?"Pay Now":"Pay After"}</h3>
      <p>Fill in your details</p>
      <input class="input" id="name" placeholder="Full Name">
      <input class="input" id="phone" placeholder="Phone Number">
      <input class="input" id="address" placeholder="Address">
      <div class="modal-actions">
        <button class="btn btn-ghost" onclick="closeModal()">Cancel</button>
        <button class="btn btn-primary" onclick="confirmOrder('${mode}')">Confirm</button>
      </div>
    `);
  },3000);
}


function confirmOrder(mode){
  alert("✅ Order placed with "+mode.toUpperCase()+" option!");
  cart={}; renderCart(); closeModal();
}


document.getElementById("pay-now").onclick=()=>verifyAndCheckout("now");
document.getElementById("pay-after").onclick=()=>verifyAndCheckout("after");


renderProducts(); renderCart();
